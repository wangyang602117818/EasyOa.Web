       Id        int     primary key identity(1,1)       not null,    --主键id
       ConName         nvarchar(10)                     not null,     --合同类型
       ConCode         tinyint      unique              not null,     --类型code
       Cdefault        bit          default(0)          not null,     --是否默认
)
