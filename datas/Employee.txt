--人事基本信息
create table Employee(
       Id        int     primary key identity(1,1)       not null,    --主键id（人员id）
       Name              nvarchar(10)                    not null,    --姓名
       QPin              varchar(30)                     not null,    --全拼
       JPin              varchar(10)                     not null,    --简拼
       Sex               char(1)                         not null,    --性别(f/m)
       Birthday          date                            not null,    --出生年月
       IdCard            varchar(20)                     null,        --身份证号
       Telephone         varchar(30)                     null,        --联系电话
       QQ                varchar(20)                     null,        --qq
       Email             varchar(30)                     null,        --邮件
       Homephone         varchar(30)                     null,        --家庭电话
       AreaCode          int                             not null,    --家庭所在地id(区域id)
       HomeAdress        nvarchar(50)                    not null,    --家庭住址
       PeopleNumber      tinyint                         null,        --家庭人口数
       IsMarried         bit                             not null,    --婚否
       StateCode         char(3)                         not null,    --国籍代码
       FamilyCode        tinyint   default(1)            not null,    --名族id
       PoliticalCode     char(2)                         not null,    --政治面貌id
       FinishSchool      nvarchar(30)                    null,        --毕业院校
       DegreeCode        tinyint                         not null,    --学历id
       Major             nvarchar(20)                    null,        --专业
       Height            tinyint                         null,        --身高cm
       Weight            tinyint                         null,        --体重kg
       BeginWorkDate     date                            null,        --参加工作时间
       LogDate           datetime     default(getdate()) not null,    --入库时间
)
--人事-家庭人员信息
create table EmpFamilyMember(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       Name                  nvarchar(10)                not null,    --姓名
       Relation              nvarchar(10)                not null,    --与本人关系
       Birthday              date                        null,        --出生年月
       IdCard                varchar(20)                 null,        --身份证号
       ContactPhone          varchar(30)                 null,        --联系电话
       ContactAddress        nvarchar(50)                null,        --联系地址
       Company               nvarchar(50)                null,        --工作单位
       PoliticalCode         char(2)                     not null,    --政治面貌id
       LogDate           datetime     default(getdate()) not null,    --入库时间
)
--人事-教育学习情况
create table EmpEducationExperience(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       EduStartDate          date                        not null,    --开始时间
       EduEndDate            date                        not null,    --结束时间
       School                nvarchar(30)                not null,    --学校
       DegreeCode            tinyint                     not null,    --学历id
       LearnContent          nvarchar(100)               null,        --学习内容
       LearnResult           nvarchar(100)               null,        --学习结果
       CredentialId          varchar(50)                 null,        --证件号码
       LogDate           datetime     default(getdate()) not null,    --入库时间
)
--人事-工作经历
create table EmpWorkExperience(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       WorkExpStartDate      date                        not null,    --开始日期
       WorkExpEndDate        date                        null,        --结束时间
       CompanyName           nvarchar(50)                null,        --公司名称
       JobTitle              nvarchar(30)                null,        --职位名称
       LeaveReason           nvarchar(200)               null,        --离职原因
       LogDate               datetime                    not null,    --录入时间
)
--人事-公司信息表
create table EmpCompany(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId       int                              not null,    --人员id
       EmpCode          int                              null,        --工号
       DepartmentCode   varchar(20)                      not null,    --部门代码(主)
       JobTitleCode     varchar(20)                      not null,    --职位代码(主)
       EmpStateCode     int                              not null,    --状态代码
       JoinDate         date                             not null,    --入职时间
       RegularDate      date                             null,        --转正时间
       LeaveDate        date                             null,        --离职时间
       Integral         int                              null,        --总培训积分
       LogDate          datetime     default(getdate())  not null,    --录入时间
)
--人事-员工部门关联表（多个部门）
create table EmpDeptRelation(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId       int                              not null,    --人员id
       DepartmentCode   varchar(20)                      not null,    --部门代码
)
--人事-员工职位关联表（多个职位）
create table EmpJobRelation(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId       int                              not null,    --人员id
       JobTitleCode     varchar(20)                      not null,    --职位代码
)
--人事-合同信息
create table EmpContract(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       ContractTypeCode      tinyint                     not null,    --合同类型代码
       ContractId            varchar(20)                 not null,    --合同编号
       ConBeginDate          date                        null,        --合同开始时间
       ConEndDate            date                        null,        --合同结束时间
       ConDepartmentCode     varchar(20)                 null,        --约定部门
       ConJobTitleCode       varchar(20)                 null,        --约定职位
       ConPay                smallmoney                  null,        --约定薪资
       RelieveDate           date                        null,        --解除合同时间
       RelieveReason         varchar(100)                null,        --解除原因
       IsUsed                bit        default(1)       not null,    --是否生效
       ModifyTime            datetime                    not null,    --最后一次修改时间
       ModifyEmpId           int                         not null,    --修改人
       LogDate               datetime                    not null,    --录入时间
)
--人事-奖惩信息
create table EmpAward(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       AwardTypeCode         smallint                    not null,    --奖惩类型代码
       Content               nvarchar(200)               not null,    --奖惩内容
       According             nvarchar(100)               not null,    --奖惩依据
       AwardDate             date                        not null,    --奖惩时间
       Integral            int                           null,        --该次奖惩积分
       ModifyEmpId           int                         not null,    --修改人
       LogDate               datetime                    not null,    --录入时间
)
--人事-调动信息
create table EmpTransfer(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       TransferTypeCode      smallint                    not null,    --调动类型代码
       OldDeptCode           varchar(20)                 not null,    --原部门
       OldJTCode             varchar(20)                 not null,    --原职位
       NewDeptCode           varchar(20)                 not null,    --新部门
       NewJTCode             varchar(20)                 not null,    --新职位
       FireDate              date                        not null,    --生效时间
       Remark                nvarchar(200)               null,        --备注
       ModifyEmpId           int                         not null,    --修改人
       LogDate            datetime  default(getdate())   not null,    --录入时间
)
--人事-人员培训记录
create table EmpLearn(
       Id        int     primary key identity(1,1)       not null,    --主键id
       EmployeeId            int                         not null,    --人员id
       EducateRecordId       int                         not null,    --培训主题id
       ModifyEmpId           int                         not null,    --修改人
       LogDate           datetime  default(getdate())    not null,    --录入时间
)
