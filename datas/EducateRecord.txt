--培训主题
create table EducateRecord(
       Id        int     primary key identity(1,1)       not null,    --主键id
       LearnTypeCode     int                             not null,    --培训类型代码
       LearnMethodCode   int                             not null,    --培训方式代码
       LTitle            nvarchar(30)                    not null,    --培训主题
       LContent          nvarchar(500)                   not null,    --培训内容
       LAddress          nvarchar(100)                   not null,    --培训地址
       IsInnerTeacher      bit                           not null,    --是否内部老师培训
       TeacherName       nvarchar(20)                    null,        --老师姓名
       LStartTime          datetime                      not null,    --开始时间
       LEndTime            datetime                      not null,    --结束时间
       Integral            int                           null,        --该次培训积分
       Remark            nvarchar(500)                   null,        --备注
)
