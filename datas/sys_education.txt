 drop table sys_education
create table sys_education(
    id          int     identity(1,1)  primary key   not null,     --主键id
    edu_name    nvarchar(20)                         not null,     --学历名称
    edu_code    tinyint    unique                    not null,     --学历代码
    c_default   bit     default(0)                   null,         --是否默认   
)
