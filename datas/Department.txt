create table Department(
        Id          int     identity(1,1)  primary key   not null,     --主键id
	De_Name     nvarchar(20)                         not null,     --部门名称
	De_Code     varchar(20) unique                   not null,     --部门代码（一个部门占2位）
	QPin        varchar(50)                          null,         --全拼
	JPin        varchar(20)                          null,         --简拼
        Sort        smallint  default(0)                 null,         --部门排序
)
